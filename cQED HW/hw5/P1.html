# --- Define the Drive Function epsilon(t) ---
def epsilon(t, T_pulse, A):
    if 0 <= t < T_pulse:
        return A
    else:
        return 0.0

# --- Define the ODE:  ---
def system_dynamics(t, alpha_out, Delta_0, kappa, chi, T_pulse, A):
    # chi is frequency shift due to the qubit state. Difference in frequency shift for |e> vs |g> is 2*chi.
    d_alpha_out = -(kappa / 2.0 + 1j * (Delta_0 + chi)) * alpha_out + np.sqrt(kappa) * epsilon(t, T_pulse, A)
    return d_alpha_out

def generate_noisy_trajectories(my_noiseless_traj, N_copies=25000, noise_intensity=0.05):
    '''Add white noise on top of some base noiseless trajectory, generate many such realizations'''
    traj_list = []
    for i in range(N_copies):
        added_noise_real = np.random.normal(loc=0,scale=noise_intensity*np.sqrt(len(my_noiseless_traj)),size=len(my_noiseless_traj)) # noise for real quadrature
        added_noise_imag = np.random.normal(loc=0,scale=noise_intensity*np.sqrt(len(my_noiseless_traj)),size=len(my_noiseless_traj)) # noise for imag quadrature
        
        traj_list.append(my_noiseless_traj + added_noise_real + 1j*added_noise_imag)
    
    traj_list = np.array(traj_list, dtype=complex)
    return traj_list

def fit_to_complex_data_to_2D_gaussian_symmetric(complex_data, bins=100, verbose=False):
    """
    Fits a 2D Gaussian (assuming sigma_x = sigma_y = sigma, which is true for us here) to the density 
    of complex data points in the IQ plane.
    
    Returns:
        center (complex valued), sigma, amplitude
    """
    # Extract real (I) and imaginary (Q) components
    I_data = np.real(complex_data)
    Q_data = np.imag(complex_data)
    
    # Find initial guesses
    mean_I = np.mean(I_data)
    mean_Q = np.mean(Q_data)
    std_I = np.std(I_data)
    std_Q = np.std(Q_data) 
    
    # Calculate a unified standard deviation guess (geometric mean)
    std_guess = np.sqrt(std_I * std_Q)

    H, X, Y = get_histogram(complex_data, bins=bins)
        
    # Flatten the grids for curve_fit
    xdata = np.vstack((X.ravel(), Y.ravel()))
    ydata = H.ravel()

    # Compile initial guesses: [amplitude, x0, y0, sigma]
    amplitude_guess = np.max(H)
    initial_guess = [amplitude_guess, mean_I, mean_Q, std_guess]
    
    # Perform the curve fit
    try:
        popt, pcov = curve_fit(gaussian_2d, xdata, ydata, p0=initial_guess)
        if verbose:
            print(f"Fit successful! Center: {popt[1]:.5f} + {popt[2]:.5f}j | Sigma: {popt[3]:.5f}")

        return (popt[1] + 1j*popt[2]), popt[3], popt[0]
    except RuntimeError:
        print("Error: Optimal parameters not found. Did you give me data for one gaussian or two?")
        return None, None, None
    
# Define the simplified, symmetric 2D Gaussian function
def gaussian_2d(xy, amplitude, x0, y0, sigma):
    x, y = xy
    # Calculate the symmetric Gaussian equation
    g = amplitude * np.exp(-((x - x0)**2 + (y - y0)**2) / (2 * sigma**2))
    return g

def get_histogram(complex_data, bins=100):
    I_data = np.real(complex_data)
    Q_data = np.imag(complex_data)
    
    H, xedges, yedges = np.histogram2d(I_data, Q_data, bins=bins)
    
    x_centers = (xedges[:-1] + xedges[1:]) / 2
    y_centers = (yedges[:-1] + yedges[1:]) / 2
    X, Y = np.meshgrid(x_centers, y_centers, indexing='ij')
    
    return H, X, Y

def qubit_decay_readout(t_decay, T_pulse, t_final, Delta_0, kappa, chi, A):
    '''
    Simulate the readout of a qubit that decays during the readout pulse. 
    t_decay: time at which the qubit decays from |1> to |0>
    T_pulse: duration of the readout pulse
    t_final: total simulation time (should be >= T_pulse)
    Delta_0, kappa, chi, A: system parameters as before
    '''
    t_eval = np.linspace(0, t_final, 500)
    initial_state = [0.0 + 0.0j]  # alpha(0) = 0

    # Define a piecewise function for chi that changes at t_decay
    def chi_time_dependent(t):
        if t < t_decay:
            return chi/2
        else:
            return -chi/2

    # Define a modified system dynamics function that incorporates the time-dependent chi
    def system_dynamics_time_dependent(t, alpha_out):
        current_chi = chi_time_dependent(t)
        d_alpha_out = -(kappa / 2.0 + 1j * (Delta_0 + current_chi)) * alpha_out + np.sqrt(kappa) * epsilon(t, T_pulse, A)
        return d_alpha_out

    sol = solve_ivp(
        fun=system_dynamics_time_dependent,
        t_span=(0, t_final),
        y0=initial_state,
        t_eval=t_eval,
        method='RK45', rtol=1e-10, atol=1e-12, max_step=1e-3) # Use tight tolerances and small max step to handle the discontinuity at t_decay

    return sol.t, sol.y[0]

        roots.append(t0)
    roots = np.array(roots)
    print("Zero-crossing t_decay values (linear interp):", roots)
    print("First crossing:", roots[0])
